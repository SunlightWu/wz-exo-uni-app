export function createThrottle(minIntervalMs) {
  let last = 0;
  return () => {
    const now = Date.now();
    if (now - last < minIntervalMs) return false;
    last = now;
    return true;
  };
}

// services/ble.js
import { SERVICE_UUID, SCAN_FILTER_UUID, CHAR_UUIDS } from '../utils/protocol.js';

// ── 初始化蓝牙适配器 ──
export function initBluetooth() {
  return new Promise((resolve, reject) => {
    uni.openBluetoothAdapter({
      success: () => {
        uni.onBluetoothAdapterStateChange((res) => {
          console.log('[BLE] 适配器状态:', res.available);
        });
        resolve();
      },
      fail: (err) => reject(new Error('蓝牙初始化失败: ' + err.errMsg)),
    });
  });
}

// ── 扫描设备（按广播 UUID 过滤） ──
export function startScan(timeoutSec = 8) {
  return new Promise((resolve, reject) => {
    const devices = [];
    uni.onBluetoothDeviceFound((res) => {
      for (const dev of res.devices) {
        if (!devices.find(d => d.deviceId === dev.deviceId)) {
          devices.push(dev);
        }
      }
    });
    uni.startBluetoothDevicesDiscovery({
      services: [SCAN_FILTER_UUID],
      allowDuplicatesKey: false,
      interval: 0,
      success: () => {
        setTimeout(() => {
          uni.stopBluetoothDevicesDiscovery();
          devices.sort((a, b) => (b.RSSI || -100) - (a.RSSI || -100));
          resolve(devices);
        }, timeoutSec * 1000);
      },
      fail: (err) => reject(err),
    });
  });
}

// ── 连接设备 ──
export async function connectDevice(deviceId) {
  await new Promise((resolve, reject) => {
    uni.createBLEConnection({ deviceId, timeout: 10000, success: resolve, fail: reject });
  });
  const { services } = await new Promise((resolve, reject) => {
    uni.getBLEDeviceServices({ deviceId, success: resolve, fail: reject });
  });
  const svc = services.find(s => s.uuid.toUpperCase().includes('FFF0'));
  if (!svc) throw new Error('未找到 FFF0 主服务');
  const { characteristics } = await new Promise((resolve, reject) => {
    uni.getBLEDeviceCharacteristics({ deviceId, serviceId: svc.uuid, success: resolve, fail: reject });
  });
  return { deviceId, serviceId: svc.uuid, characteristics };
}

// ── 全局 Notify 监听 ──
let onNotifyData = null;
export function registerNotifyHandler(handler) {
  if (!onNotifyData) {
    onNotifyData = handler;
    uni.onBLECharacteristicValueChange((res) => {
      onNotifyData(res.characteristicId, res.value);
    });
  } else {
    onNotifyData = handler;
  }
}

// ── 订阅特征值 Notify ──
export function subscribeNotify(deviceId, charId) {
  return new Promise((resolve, reject) => {
    uni.notifyBLECharacteristicValueChange({
      deviceId, serviceId: SERVICE_UUID, characteristicId: charId,
      state: true,
      success: resolve,
      fail: reject,
    });
  });
}

// ── 写入（串行队列，Android notify 后立即 write 需延时 50ms） ──
const writeQueue = [];
let writing = false;
async function processWriteQueue() {
  if (writing || writeQueue.length === 0) return;
  writing = true;
  const { deviceId, charId, buffer, resolve, reject } = writeQueue.shift();
  await new Promise(r => setTimeout(r, 50)); // Android 延时
  try {
    await new Promise((res, rej) => {
      uni.writeBLECharacteristicValue({
        deviceId, serviceId: SERVICE_UUID, characteristicId: charId,
        value: buffer,
        writeType: 'write', // 外骨骼协议必须用回复写
        success: res,
        fail: rej,
      });
    });
    resolve();
  } catch (e) {
    reject(e);
  } finally {
    writing = false;
    processWriteQueue();
  }
}
export function writeCharacteristic(deviceId, charId, buffer) {
  return new Promise((resolve, reject) => {
    writeQueue.push({ deviceId, charId, buffer, resolve, reject });
    processWriteQueue();
  });
}

// ── 连接状态监听 ──
export function onConnectionStateChange(callback) {
  uni.onBLEConnectionStateChange((res) => {
    callback(res.connected, res.deviceId);
  });
}

// ── 断开 ──
export function disconnect(deviceId) {
  return new Promise((resolve) => {
    uni.closeBLEConnection({ deviceId, complete: resolve });
  });
}

// ── 关闭适配器 ──
export function closeBluetooth() {
  uni.closeBluetoothAdapter();
}

### 3.3 文件 3：`services/exo-command.js`

> 命令服务 + ACK 互斥锁。实现与 Flutter `sendCommandAndWaitAck` 完全等价的功能。

```javascript
// services/exo-command.js
import { CHAR_UUIDS, encodeCommand } from '../utils/protocol.js';
import { writeCharacteristic } from './ble.js';

const CMD_CHAR = CHAR_UUIDS.command;

// ── ACK 互斥锁 ──
export class AckLock {
  constructor() {
    this._pending = null;
    this._resolve = null;
  }
  acquire() {
    if (this._pending) throw new Error('ACK 通道被占用');
    this._pending = new Promise((resolve) => { this._resolve = resolve; });
    return this._pending;
  }
  release() {
    if (this._resolve) this._resolve();
    this._pending = null;
    this._resolve = null;
  }
}

let pendingResolve = null, pendingReject = null, ackTimeout = null;

export function handleAckByte(byte) {
  if (!pendingResolve) return;
  clearTimeout(ackTimeout);
  const res = pendingResolve, rej = pendingReject;
  pendingResolve = null;
  pendingReject = null;
  const messages = {
    0x00: null, 0x01: '运行中忙', 0x02: '状态不允许',
    0x03: '故障中', 0x04: '参数无效', 0x05: '电机未就绪', 0x06: '超时',
  };
  const msg = messages[byte];
  if (msg) rej(new Error(msg));
  else res();
}

const lock = new AckLock();

export async function sendCommand(deviceId, code, payload = {}, timeoutMs = 3000) {
  await lock.acquire();
  try {
    const buf = encodeCommand(code, payload);
    if (!buf || buf.byteLength === 0) {
      lock.release();
      return;
    }
    const ackP = new Promise((resolve, reject) => {
      pendingResolve = resolve;
      pendingReject = reject;
      ackTimeout = setTimeout(() => {
        pendingResolve = null;
        pendingReject = null;
        reject(new Error('指令响应超时'));
      }, timeoutMs);
    });
    await writeCharacteristic(deviceId, CMD_CHAR, buf);
    await ackP;
  } finally {
    lock.release();
  }
}
```

### 3.4 文件 4：`store/device.js`

> Pinia 设备状态管理，对应 Flutter Riverpod 的 Provider 体系。

```javascript
// store/device.js
import { defineStore } from 'pinia';
import { ref, computed } from 'vue';

export const useDeviceStore = defineStore('device', () => {
  // ── 基础状态 ──
  const deviceId = ref(null);
  const deviceName = ref('');
  const connected = ref(false);

  // ── 模式 ──
  const mode = ref('transparent');   // transparent | assist | training
  const assistLevel = ref(3);
  const impedanceLevel = ref(3);

  // ── 运行状态 ──
  const outputState = ref('idle');    // idle | starting | running | stopping
  const enabled = ref(false);
  const controlsLocked = ref(false);
  const isStopping = ref(false);
  const commandLoading = ref(new Set());  // 记录正在执行的命令

  // ── 设备状态（来自 FFF2 解析） ──
  const status = ref({
    running: false,
    battery: 0,
    errorCode: 0,
    leftHipAngle: 0,
    rightHipAngle: 0,
    leftTorqueNm: 0,
    rightTorqueNm: 0,
    assistLevel: 0,
    impedanceLevel: 0,
    stepCount: 0,
    runtimeSeconds: 0,
  });

  // ── 冷却态（防止 FFF2 回推覆盖用户操作） ──
  const modeCooldown = ref(false);
  const levelCooldown = ref(false);

  // ── 计算属性 ──
  const activeLevel = computed(() =>
    mode.value === 'training' ? impedanceLevel.value : assistLevel.value
  );

  // ── Actions ──
  function setDevice(id, name) {
    deviceId.value = id;
    deviceName.value = name;
    connected.value = true;
  }

  function setConnected(val) {
    connected.value = val;
    if (!val) {
      outputState.value = 'idle';
      enabled.value = false;
    }
  }

  function setMode(m) { mode.value = m; }
  function setOutputState(s) { outputState.value = s; }
  function setEnabled(v) { enabled.value = v; }
  function setControlsLocked(v) { controlsLocked.value = v; }

  function addCommandLoading(key) { commandLoading.value.add(key); }
  function removeCommandLoading(key) { commandLoading.value.delete(key); }

  function updateStatus(data) {
    // 仅在非冷却态时同步设备端的状态回推
    if (!modeCooldown.value) {
      status.value.mode = data.mode;
    }
    if (!levelCooldown.value) {
      status.value.assistLevel = data.assistLevel;
      status.value.impedanceLevel = data.impedanceLevel;
    }
    status.value.running = data.running;
    status.value.battery = data.battery;
    status.value.errorCode = data.errorCode;
    status.value.leftHipAngle = data.leftHipAngle;
    status.value.rightHipAngle = data.rightHipAngle;
    status.value.leftTorqueNm = data.leftTorqueNm;
    status.value.rightTorqueNm = data.rightTorqueNm;
    status.value.stepCount = data.stepCount;
    status.value.runtimeSeconds = data.runtimeSeconds;

    // 状态机转换
    if (data.running && outputState.value === 'starting') {
      outputState.value = 'running';
      enabled.value = true;
    }
    if (!data.running && outputState.value === 'stopping') {
      outputState.value = 'idle';
      isStopping.value = false;
    }
  }

  function setModeCooldown() {
    modeCooldown.value = true;
    setTimeout(() => { modeCooldown.value = false; }, 2000);
  }

  function setLevelCooldown() {
    levelCooldown.value = true;
    setTimeout(() => { levelCooldown.value = false; }, 2000);
  }

  function clearDevice() {
    deviceId.value = null;
    deviceName.value = '';
    connected.value = false;
    outputState.value = 'idle';
    enabled.value = false;
    controlsLocked.value = false;
    isStopping.value = false;
    commandLoading.value = new Set();
  }

  return {
    deviceId, deviceName, connected,
    mode, assistLevel, impedanceLevel,
    outputState, enabled, controlsLocked, isStopping, commandLoading,
    status, modeCooldown, levelCooldown,
    activeLevel,
    setDevice, setConnected, setMode, setOutputState, setEnabled,
    setControlsLocked, addCommandLoading, removeCommandLoading,
    updateStatus, setModeCooldown, setLevelCooldown, clearDevice,
  };
});
```

// services/device-service.js
import { useDeviceStore } from '../store/device.js';
import { sendCommand } from './exo-command.js';
import { subscribeNotify, registerNotifyHandler } from './ble.js';
import { CMD, CHAR_UUIDS, parseFFF2, parseFFF4, parseFFF5, handleAckByte } from '../utils/protocol.js';

let deviceId = null;

// ── 注册 Notify 分发 ──
export function setupNotifyHandlers() {
  registerNotifyHandler((charId, value) => {
    const store = useDeviceStore();
    const buffer = typeof value === 'string' ? uni.base64ToArrayBuffer(value) : value;

    switch (charId) {
      case CHAR_UUIDS.status: {      // FFF2
        const data = parseFFF2(buffer);
        if (data) {
          // 第一字节可能是 ACK
          const firstByte = new Uint8Array(buffer)[0];
          if (firstByte <= 0x06 && data.stateCode !== firstByte) {
            handleAckByte(firstByte);
          }
          store.updateStatus(data);
        }
        break;
      }
      case CHAR_UUIDS.fault: {       // FFF4
        const fault = parseFFF4(buffer);
        if (fault && fault.errorCode !== 0) {
          store.controlsLocked = true;
        }
        break;
      }
      case CHAR_UUIDS.realtime: {    // FFF5
        const sensor = parseFFF5(buffer);
        // 传感器数据处理（可更新到 store 或直接用于 UI）
        break;
      }
    }
  });
}

// ── 连接并配置设备（完整链路） ──
export async function connectAndConfigure(dev, userProfile = {}) {
  const store = useDeviceStore();
  const { deviceId: id, characteristics } = dev;
  deviceId = id;

  // 订阅所有 Notify 特征
  const notifyChars = characteristics.filter(c => c.properties.notify);
  for (const ch of notifyChars) {
    await subscribeNotify(deviceId, ch.uuid);
  }

  store.setDevice(deviceId, dev.name || 'WZ_EXO_S3');

  // 1. hello 握手
  await sendCommand(deviceId, CMD.hello);

  // 2. 下发用户参数
  await sendCommand(deviceId, CMD.setUserProfile, userProfile);

  // 3. 下发保存的模式
  await sendCommand(deviceId, CMD.setMode, { mode: store.mode });

  // 4. 下发等级（仅非透明模式）
  if (store.mode !== 'transparent') {
    await sendCommand(deviceId, CMD.setAssistLevel, { level: store.assistLevel });
    await sendCommand(deviceId, CMD.setImpedanceLevel, { level: store.impedanceLevel });
  }

  return deviceId;
}

// ── 开始运行 ──
export async function startRunning() {
  const store = useDeviceStore();
  if (store.outputState === 'starting' || store.outputState === 'stopping') return;
  if (store.commandLoading.size > 0) return;

  store.setOutputState('starting');
  store.addCommandLoading('start');

  try {
    await sendCommand(deviceId, CMD.start);
    // 800ms 兜底定时器
    setTimeout(() => {
      if (store.outputState === 'starting') {
        if (store.status.running) {
          store.setOutputState('running');
          store.setEnabled(true);
        } else {
          store.setOutputState('idle');
          store.setEnabled(false);
        }
      }
    }, 800);
  } catch (e) {
    store.setOutputState('idle');
    throw e;
  } finally {
    store.removeCommandLoading('start');
  }
}

// ── 停止运行 ──
export async function stopRunning() {
  const store = useDeviceStore();
  if (store.outputState === 'stopping') return;

  store.setOutputState('stopping');
  store.addCommandLoading('stop');

  try {
    await sendCommand(deviceId, CMD.stop);
    store.setEnabled(false);
    store.isStopping = true;
    // 3 秒兜底
    setTimeout(() => {
      if (store.outputState === 'stopping') {
        store.setOutputState('idle');
        store.isStopping = false;
      }
    }, 3000);
  } catch (e) {
    store.setOutputState('running');
    store.setEnabled(true);
    throw e;
  } finally {
    store.removeCommandLoading('stop');
  }
}

// ── 切换模式（待机状态） ──
export async function switchMode(targetMode) {
  const store = useDeviceStore();
  store.addCommandLoading('modeSwitch');
  try {
    await sendCommand(deviceId, CMD.setMode, { mode: targetMode });
    store.setMode(targetMode);
    store.setModeCooldown();
  } finally {
    store.removeCommandLoading('modeSwitch');
  }
}

// ── 切换等级（带防抖） ──
let debounceTimer = null;
export function changeLevel(level, isTraining) {
  const store = useDeviceStore();
  clearTimeout(debounceTimer);
  debounceTimer = setTimeout(async () => {
    if (store.commandLoading.size > 0) return;
    store.addCommandLoading('levelChange');
    try {
      const code = isTraining ? CMD.setImpedanceLevel : CMD.setAssistLevel;
      await sendCommand(deviceId, code, { level });
      if (isTraining) store.impedanceLevel = level;
      else store.assistLevel = level;
      store.setLevelCooldown();
    } finally {
      store.removeCommandLoading('levelChange');
    }
  }, 300);
}

// ── 清除故障 ──
export async function clearFault() {
  await sendCommand(deviceId, CMD.clearFault);
}

// ── 断开 ──
export async function disconnectDevice() {
  if (deviceId) {
    await disconnect(deviceId);
    deviceId = null;
  }
}

// services/api.js
const BASE_URL = 'https://your-api-domain.com';

async function request(method, path, data = null) {
  const token = uni.getStorageSync('token');
  return new Promise((resolve, reject) => {
    uni.request({
      url: `${BASE_URL}${path}`,
      method,
      data,
      header: {
        'Authorization': `Bearer ${token}`,
        'Content-Type': 'application/json',
      },
      success: (res) => {
        if (res.statusCode === 200) resolve(res.data);
        else reject(new Error(res.data.message || '请求失败'));
      },
      fail: reject,
    });
  });
}

import { api } from './api.js';

export async function wxLogin() {
  return new Promise((resolve, reject) => {
    uni.login({
      success: async (res) => {
        try {
          const result = await api.wxLogin(res.code);
          uni.setStorageSync('token', result.token);
          uni.setStorageSync('userInfo', result.user);
          resolve(result);
        } catch (e) {
          reject(e);
        }
      },
      fail: reject,
    });
  });
}

## 5. uni-app 开发注意事项

| 要点 | 说明 |
|------|------|
| 条件编译 | BLE API 用 `#ifdef MP-WEIXIN`，确保只在小程序端编译 |
| 地图组件 | 使用 `<map>` 组件，需在 `manifest.json` 配置地图 SDK（腾讯/高德） |
| 支付 | 使用 `uni.requestPayment`，需后端预下单获取支付参数 |
| 扫码 | 使用 `uni.scanCode({ onlyFromCamera: true })`，禁止从相册扫码 |
| 蓝牙权限 | `manifest.json` 中声明 `scope.bluetooth`，用户首次使用时弹授权 |
| 蓝牙写入 | `writeType: 'write'`（强制回复写），Android 需 notify 后延时 50ms 再 write |
| 后台保活 | 控制页 `onHide` 不关蓝牙，`onShow` 检测重连；小程序切后台 5 分钟后可能被挂起 |
| 数据持久化 | 用户参数和偏好用 `uni.setStorageSync`，运动数据实时上传服务端 |
| 安全 | 二维码做签名校验（防伪造设备 SN），令牌校验每次 BLE 连接都做 |
